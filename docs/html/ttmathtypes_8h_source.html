557 class ExceptionInfo
558 {
559 const char * file;
560 int line;
561
562 public:
563 ExceptionInfo() : file(0), line(0) {}
564 ExceptionInfo(const char * f, int l) : file(f), line(l) {}
565
566 std::string Where() const
567 {
568 if( !file )
569 return "unknown";

571 std::ostringstream result;
