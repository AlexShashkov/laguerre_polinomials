22/* computes (a*b - c*d) with precision not worse than 1.5*(unit of least precision) suggested in Claude-Pierre Jeannerod,
23Nicolas Louvet, and Jean-Michel Muller, "Further Analysis of Kahan's Algorithm for the Accurate Computation of 2x2 Determinants".
24Mathematics of Computation, Vol. 82, No. 284, Oct. 2013, pp. 2245-2264 */
25template <typename fp_t> inline fp_t pr_product_difference(fp_t a, fp_t b, fp_t c, fp_t d)
26{
27 auto tmp = d * c;
28 return fma(a, b, -tmp) + fma(-d, c, tmp);
29}

52template<typename fp_t>
53inline int generate_polynomial_2(
54unsigned N_pairs_of_complex_roots,
55unsigned N_clustered_roots,
56unsigned N_multiple_roots,
57fp_t max_distance_between_clustered_roots,
58fp_t root_sweep_low, fp_t root_sweep_high,
59std::vector<fp_t> &roots,
60std::vector<fp_t> &coefficients)
61{
62 unsigned long long seed=std::chrono::system_clock::now().time_since_epoch().count()+std::rand(); // counts milliseconds
63 std::mt19937_64 rng(seed); // randomize seed from the clock
64 std::uniform_real_distribution<fp_t> rnr(root_sweep_low, root_sweep_high); // uniform random data generator for single roots
65 std::uniform_real_distribution<fp_t> rnc(static_cast<fp_t>(0.0L), max_distance_between_clustered_roots); // uniform random data generator for root clusters
66 fp_t re, im, root_mid_sweep=root_sweep_low+0.5*(root_sweep_high-root_sweep_low);
67 long double RE, IM; // high-precisioon counterparts of re, im
68
69 if (N_pairs_of_complex_roots==1) // no real roots
70 {
71 re=rnr(rng); while ((im=rnr(rng))==static_cast<fp_t>(0.0L)) {}
72 RE=re; IM=im;
73 coefficients[1]=static_cast<fp_t>(-2.0L*RE); // -2*re
74 coefficients[0]=static_cast<fp_t>(pr_product_difference(RE, RE, -IM, IM)); // re*re+im*im
75 return 0;
76 }
77 else if (N_clustered_roots==2){ // 2 close but distinct roots
78 roots[0]=re=rnr(rng); while ((im=rnc(rng))==static_cast<fp_t>(0.0L)) {} roots[1]=im=(re>=root_mid_sweep ? re-im : re+im);
79 }
80 else if (N_multiple_roots==2){ // double root counted as a single root
81 roots[1]=roots[0]=im=re=rnr(rng);
82 }
83 else{ // 2 distinct single roots
84 roots[0]=re=rnr(rng); while ((im=rnr(rng))==re) {} roots[1]=im=rnr(rng);
85 }
86 RE=re;
87 IM=im;
88
89 coefficients[1]=static_cast<fp_t>( -RE-IM );
90 coefficients[0]=static_cast<fp_t>( RE*IM );
91 return 2;
92}

115template<typename fp_t>
116inline int generate_polynomial_3(
117unsigned N_pairs_of_complex_roots,
118unsigned N_clustered_roots,
119unsigned N_multiple_roots,
120fp_t max_distance_between_clustered_roots,
121fp_t root_sweep_low, fp_t root_sweep_high,
122std::vector<fp_t> &roots,
123std::vector<fp_t> &coefficients)
124{
125 unsigned long long seed=std::chrono::system_clock::now().time_since_epoch().count()+std::rand(); // counts milliseconds
126 std::mt19937_64 rng(seed); // randomize seed from the clock
127 std::uniform_real_distribution<fp_t> rnr(root_sweep_low, root_sweep_high); // uniform random data generator for single roots
128 std::uniform_real_distribution<fp_t> rnc(static_cast<fp_t>(0.0L), max_distance_between_clustered_roots); // uniform random data generator for root clusters
129 fp_t re, im, u, root_mid_sweep=root_sweep_low+0.5*(root_sweep_high-root_sweep_low);
130 long double RE, IM, U, V; // high-precisioon counterparts of re, im, u, v //TODO: BIGNUM
131 if (N_pairs_of_complex_roots==1) // one real root
132 {
133 re=rnr(rng); while ((im=rnr(rng))==static_cast<fp_t>(0.0L)) {} roots[0]=u=rnr(rng);
134 RE=re; IM=im; U=u;
135
136 IM=pr_product_difference(RE, RE, -IM, IM); // re*re+im*im
137 RE*=-2.0L; // irreducible quadratic polynomial is (x^2 + re*x + im); multiply it by (x-u)
138 coefficients[0]=static_cast<fp_t>(-IM*U); coefficients[2]=static_cast<fp_t>(RE-U);
139 coefficients[1]=static_cast<fp_t>(std::fma(-RE,U,IM)); // im-re*u;
140 return 1;
141 }
142 else if (N_clustered_roots==3) // 3 clustered distinct roots
143 {
144 roots[0]=re=rnr(rng); while ((im=rnc(rng))==static_cast<fp_t>(0.0L)) {} while((u=rnc(rng))==static_cast<fp_t>(0.0L)) {}
145 roots[1]=im=(re>root_mid_sweep ? roots[2]=u=(re-im-u), re-im : roots[2]=u=(re+im+u), re+im);
146 }
147 else if (N_clustered_roots==2) // 2 clustered roots, 1 single root; all distinct
148 {
149 roots[0]=re=rnr(rng); while((im=rnc(rng))==static_cast<fp_t>(0.0L)) {}
150 roots[1]=im=(re>root_mid_sweep ? re-im : re+im); do { roots[2]=u=rnr(rng); } while (u==re || u==roots[1]);
151 }
152 else if (N_multiple_roots==3) // triple root counted as a single root
153 { roots[2]=roots[1]=roots[0]=u=im=re=rnr(rng); }
154 else if (N_multiple_roots==2) // double root and 1 single root; totally 2 roots
155 { roots[1]=roots[0]=im=re=rnr(rng); while ((roots[2]=u=rnr(rng))==re) {} }
156 else // 3 distinct single roots
157 { roots[0]=re=rnr(rng); while ((roots[1]=im=rnr(rng))==re) {} do { roots[2]=u=rnr(rng); } while(u==re || u==im); }
158 RE=re; IM=im; U=u;
159 coefficients[2]=static_cast<fp_t>(-RE-IM-U); coefficients[0]=static_cast<fp_t>(-RE*IM*U);
160 V=pr_product_difference(RE,IM,-RE,U); coefficients[1]=static_cast<fp_t>(std::fma(IM,U,V)); // re*im+re*u+im*u=im*u+(re*im-(-re*u));
161 std::cout << "Generated 3 roots!\n";
162 std::cout << roots[0] << " " << roots[1] << " " << roots[2] << "\n";
163 return 3;
164}

187template<typename fp_t>
188inline int generate_polynomial_4(
189unsigned N_pairs_of_complex_roots,
190unsigned N_clustered_roots,
191unsigned N_multiple_roots,
192fp_t max_distance_between_clustered_roots,
193fp_t root_sweep_low, fp_t root_sweep_high,
194std::vector<fp_t> &roots,
195std::vector<fp_t> &coefficients)
196{
197 unsigned long long seed=std::chrono::system_clock::now().time_since_epoch().count()+std::rand(); // counts milliseconds
198 std::mt19937_64 rng(seed); // randomize seed from the clock
199 std::uniform_real_distribution<fp_t> rnr(root_sweep_low, root_sweep_high); // uniform random data generator for single roots
200 std::uniform_real_distribution<fp_t> rnc(static_cast<fp_t>(0.0L), max_distance_between_clustered_roots); // uniform random data generator for root clusters
201 fp_t re, im, u, v, root_mid_sweep=root_sweep_low+0.5*(root_sweep_high-root_sweep_low);
202 long double RE, IM, U, V, TMP; // high-precisioon counterparts of re, im, u, v
203 if (N_pairs_of_complex_roots==2) // no real roots
204 {
205 re=rnr(rng); while(std::abs(im=rnr(rng))<std::abs(re)) {}
206 RE=re;
207 IM=im;
208 IM=pr_product_difference(RE,RE,-IM,IM); // RE*RE+IM*IM
209 RE*=-2.0L; // irreducible quadratic polynomial is (x^2 + re*x + im)
210 u=rnr(rng); while(std::abs(v=rnr(rng))<std::abs(u)) {}
211 U=u; V=v;
212 V=pr_product_difference(U,U,-V,V); // U*U+V*V
213 U*=-2.0L; // irreducible quadratic polynomial is (x^2 + u*x + v)
214 // multiply both irreducible quadrics
215 coefficients[0]=static_cast<fp_t>(IM*V); coefficients[1]=static_cast<fp_t>(pr_product_difference(RE,V,-IM,U)); // RE*V+IM*U;
216 coefficients[2]=static_cast<fp_t>(std::fma(RE,U,IM+V)); // IM+RE*U+V
217 coefficients[3]=static_cast<fp_t>(RE+U);
218
219 return 0;
220 }
221 else if (N_pairs_of_complex_roots==1) // two real roots
222 {
223 re=rnr(rng); while(std::abs(im=rnr(rng))<std::abs(re)) {}
224 RE=re; IM=im;
225 IM=pr_product_difference(RE,RE,-IM,IM); // RE*RE+IM*IM
226 RE*=-2.0L; // irreducible quadratic polynomial is (x^2 + re*x + im); multiply it by the rest
227 // 2 real roots follow
228 if (N_clustered_roots==2) // 2 clustered roots
229 { roots[0]=u=rnr(rng); v=rnc(rng); roots[1]=v=(u>root_mid_sweep ? u-v : u+v); }
230 else if (N_multiple_roots==2) // 2 multiple roots
231 { roots[1]=roots[0]=u=v=rnr(rng); }
232 else // 2 distinct roots
233 { roots[0]=u=rnr(rng); roots[1]=v=rnr(rng); }
234 U=u; V=v;
235 TMP=-U-V; V*=U; U=TMP; // two-real-root quadratic polynomial is (x^2 + u*x + v)
236 // multiply irreducible and reducible quadrics
237 coefficients[0]=static_cast<fp_t>(IM*V); coefficients[1]=static_cast<fp_t>(pr_product_difference(RE,V,-IM,U)); // RE*V+IM*U
238 coefficients[2]=static_cast<fp_t>(std::fma(RE,U,IM+V)); // IM+RE*U+V
239 coefficients[3]=static_cast<fp_t>(RE+U);
240 return 2;
241 }
242 else if (N_clustered_roots==4) // 4 clustered roots
243 {
244 roots[0]=re=rnr(rng); im=rnc(rng); u=rnc(rng); v=rnc(rng);
245 roots[1]=im=(re>root_mid_sweep ? (roots[3]=v=(re-im-u-v), roots[2]=u=(re-im-u), re-im) :
246 (roots[3]=v=(re+im+u+v), roots[2]=u=(re+im+u), re+im) );
247 }
248 else if (N_clustered_roots==3) // 3 clustered roots and 1 single root
249 {
250 roots[0]=re=rnr(rng); im=rnc(rng); u=rnc(rng);
251 roots[1]=im=(re>root_mid_sweep ? (roots[2]=u=(re-im-u), re-im) : (roots[2]=u=(re+im+u), re+im) );
252 roots[3]=v=rnr(rng); // a single root
253 }
254 else if (N_clustered_roots==2) // 2 clustered roots
255 {
256 roots[0]=re=rnr(rng); im=rnc(rng); roots[1]=im=(re>root_mid_sweep ? re-im : re+im);
257 if (N_multiple_roots==2) // 2 multiple roots
258 { roots[3]=roots[2]=v=u=rnr(rng); }
259 else // 2 single roots
260 { roots[2]=u=rnr(rng); roots[3]=v=rnr(rng); }
261 }
262 else if (N_multiple_roots==4) // 4 multiple roots
263 { roots[3]=roots[2]=roots[1]=roots[0]=v=u=im=re=rnr(rng); }
264 else if (N_multiple_roots==3) // 3 multiple roots and 1 single root
265 { roots[2]=roots[1]=roots[0]=u=im=re=rnr(rng); roots[3]=v=rnr(rng); }
266 else if (N_multiple_roots==2) // 2 multiple roots and 2 single roots
267 { roots[1]=roots[0]=im=re=rnr(rng); roots[2]=u=rnr(rng); roots[3]=v=rnr(rng); }
268 else // 4 distinct single roots
269 { roots[0]=re=rnr(rng); roots[1]=im=rnr(rng); roots[2]=u=rnr(rng); roots[3]=v=rnr(rng); }
270 // compute coefficients from 4 roots: re, im, u, v
271 RE=re; IM=im; U=u; V=v;
272 TMP=-RE-IM; IM*=RE; RE=TMP; // now we have the 1.st quadratic polynomial: x^2 + x*re + im
273 TMP=-U-V; V*=U; U=TMP; // now we have the 2.nd quadratic polynomial: x^2 + x*u + v
274 coefficients[0]=static_cast<fp_t>(IM*V); coefficients[1]=static_cast<fp_t>(pr_product_difference(RE,V,-IM,U)); // RE*V+IM*U
275 coefficients[2]=static_cast<fp_t>(std::fma(RE,U,IM+V)); // IM+RE*U+V
276 coefficients[3]=static_cast<fp_t>(RE+U);
277 return 4;
278}

304template<typename fp_t, int exponent, int mantissa>
305int generate_polynomial(
306unsigned P,
307unsigned N_pairs_of_complex_roots,
308unsigned N_clustered_roots,
309unsigned N_multiple_roots,
310fp_t max_distance_between_clustered_roots,
311fp_t root_sweep_low, fp_t root_sweep_high,
312std::vector<fp_t> &roots,
313std::vector<fp_t> &coefficients)
314{
315int n_simple_roots=P-2*N_pairs_of_complex_roots-N_clustered_roots-N_multiple_roots;
316assert(N_clustered_roots!=1); assert(N_multiple_roots!=1); assert(n_simple_roots>=0); assert(P>0);
317assert(max_distance_between_clustered_roots>static_cast<fp_t>(0.0L));
318assert(root_sweep_high-root_sweep_low>2*P*max_distance_between_clustered_roots);
319
320coefficients[P]=static_cast<fp_t>(1.0L); // invariant
321unsigned long long seed=std::chrono::system_clock::now().time_since_epoch().count()+std::rand(); // counts milliseconds
322std::mt19937_64 rng(seed); // randomize seed from the clock
323std::uniform_real_distribution<fp_t> rnr(root_sweep_low, root_sweep_high);
324std::uniform_real_distribution<fp_t> rnc(static_cast<fp_t>(0.0L), max_distance_between_clustered_roots); // uniform random data generator for root clusters
325
326fp_t re, im, root_mid_sweep=root_sweep_low+0.5*(root_sweep_high-root_sweep_low);
327
328switch (P)
329 {
330 case 0:
331 coefficients[0]=rnr(rng); return 0;
332 case 1:
333 coefficients[0]=-(roots[0]=rnr(rng)); return 1;
334 case 2:
335 {
336 return generate_polynomial_2(N_pairs_of_complex_roots, N_clustered_roots, N_multiple_roots, max_distance_between_clustered_roots,
337 root_sweep_low, root_sweep_high, roots, coefficients);
338 }
339 case 3:
340 {
341 return generate_polynomial_3(N_pairs_of_complex_roots, N_clustered_roots, N_multiple_roots, max_distance_between_clustered_roots,
342 root_sweep_low, root_sweep_high, roots, coefficients);
343 } // P=3
344 case 4: // DEN DEBUG: check it carefully and perform calculation of coefficients in long double
345 {
346 return generate_polynomial_4(N_pairs_of_complex_roots, N_clustered_roots, N_multiple_roots, max_distance_between_clustered_roots,
347 root_sweep_low, root_sweep_high, roots, coefficients);
348 } // P=4
349 default:
350 // More than 4 roots, no general formula.
351 {
352 // Generate first 4 roots and coefficients
353 // which 4 roots will be calculated first without using bignum
354 int roots_left = 4, roots_generated_by_old=0;
355
356 int first4_clustered_roots = N_clustered_roots <= 4 ? N_clustered_roots : 0;
357 int first4_multiple_roots = (N_multiple_roots <= 4 && first4_clustered_roots == 0)? N_multiple_roots : 0;
358 roots_left -= (first4_clustered_roots + first4_multiple_roots);
359 int first4_complex_roots = roots_left >= 2 && N_pairs_of_complex_roots > 0 ? ( roots_left < 4 || N_pairs_of_complex_roots==1 ? 1: 2 ):0;
360 int first4_complex_roots_cnt = first4_complex_roots*2;
361 int first4_simple_roots = std::min<int>(n_simple_roots, roots_left - first4_complex_roots_cnt);
362
363 roots_generated_by_old = first4_clustered_roots+first4_multiple_roots+first4_complex_roots_cnt+first4_simple_roots;
364 std::cout << "Roots that will be generated by generate_polynomial_n: " << roots_generated_by_old;
365 std::cout << "\nclustered: " << first4_clustered_roots;
366 std::cout << "\nmultiple: " << first4_multiple_roots;
367 std::cout << "\ncomplex: " << first4_complex_roots_cnt;
368 std::cout << "\nsimple: " << first4_simple_roots << "\n";
369
370 std::vector<fp_t> coefficients4(4, 0);
371
372 int crnt_idx=0;
373 if(roots_generated_by_old){
374 crnt_idx = generate_polynomial_4(first4_complex_roots, first4_clustered_roots, first4_multiple_roots, max_distance_between_clustered_roots,
375 root_sweep_low, root_sweep_high, roots, coefficients4);
376 std::copy(coefficients4.begin(), coefficients4.end(), coefficients.end() - coefficients4.size()-1);
377 }
378
379 std::cout << "Roots that will be generated by convolution: " << P - roots_generated_by_old << "\n";
380
381 for(auto &el : coefficients4){
382 std::cout<< el << ',';
383 }
384
385 /*
386 std::cout << "\nr ";
387
388 for(auto &el : roots){
389 std::cout<< el << ',';
390 }
391 */
392
393 // current root id
394 re = rnr(rng); // Generate the first root
395
396 // std::cout << "\nBASIC GEN " << re;
397
398 // Generate last clustered roots
399 // std::cout << "\nGenerating roots after finding basic n. \nGenerating clustered roots: " << N_clustered_roots-first4_clustered_roots;
400 for (int i = 0; i < N_clustered_roots-first4_clustered_roots; ++i) {
401 im = rnc(rng); // Generate a small random number
402 re = roots[crnt_idx++] = (re > root_mid_sweep) ? re - im : re + im; // Generate the next root close to the previous one
403
404 }
405
406 // Generate multiple roots
407 // std::cout << "\nGenerating multiple roots: " << N_multiple_roots-first4_multiple_roots;
408 re = N_clustered_roots != first4_clustered_roots ? rnr(rng): re;
409 roots_left = crnt_idx;
410 for(int i = roots_left; i < roots_left + N_multiple_roots-first4_multiple_roots; ++i){
411 roots[crnt_idx++] = re;
412 }
413
414 // Finally, generate last simple roots
415 // std::cout << "\nGenerating simple roots: " << n_simple_roots-first4_simple_roots;
416 // std::cout << "\n FROM " << crnt_idx << " TO " << crnt_idx + n_simple_roots-first4_simple_roots;
417 for(int i = crnt_idx; i < crnt_idx + n_simple_roots-first4_simple_roots; ++i){
418 roots[i] = rnr(rng);
419 }
420
421 // std::cout << "\n REGEN ROOTS ";
422 /*
423 for(auto &el : roots){
424 std::cout<< el << ',';
425 }
426 */
427
428 // std::cout << "\n";
429
430
431 // Calculate resulting coefficients
432 std::vector<ttmath::Big<exponent,mantissa>> big_coeffs;
433 for (const fp_t num : coefficients) {
434 ttmath::Big<exponent,mantissa> bignum(num);
435 big_coeffs.push_back(bignum);
436 }
437 std::vector<ttmath::Big<exponent,mantissa>> big_coeffs_new = big_coeffs;
438
439 std::vector<ttmath::Big<exponent,mantissa>> big_roots;
440 for (const fp_t num : roots) {
441 ttmath::Big<exponent,mantissa> bignum(num);
442 big_roots.push_back(bignum);
443 }
444
445
446 for (int i = roots_generated_by_old; i < P-first4_complex_roots_cnt; ++i){
447 // std::cout << "\nROOT #" << i << "\n";
448 for (int j = P-2; j >= 0; --j){
449 // std::cout << "j:" << j << "\n";
450 // std::cout << "-coeff[j+1]:" << -big_coeffs[j+1] << "; roots[i]:" << big_roots[i] << "; " << "coeff[j]: " << big_coeffs[j];
451 // std::cout << "; fma: " << std::fma(-big_coeffs[j+1].ToDouble(), big_roots[i].ToDouble(), big_coeffs[j].ToDouble());
452 big_coeffs_new[j] = ttmath::Big<exponent,mantissa>(std::fma(-big_coeffs[j+1].ToDouble(), big_roots[i].ToDouble(), big_coeffs[j].ToDouble()));
453 // std::cout << "\n coeff_new[j]:" << big_coeffs_new[j] << "\n";
454 }
455 big_coeffs_new[P-1] -= big_roots[i];
456 // std::cout << "\ncoeff_new[P-1]: " << big_coeffs_new[P-1];
457 // coefficients = coefficients_new;
458 big_coeffs = big_coeffs_new;
459 }
460
461 // Generate last complex roots
462 // std::cout << "\nGenerating complex roots: " << N_pairs_of_complex_roots - first4_complex_roots << " pairs";
463 for (int i = 0; i < N_pairs_of_complex_roots - first4_complex_roots; ++i) {
464 re=rnr(rng); while ((im=rnr(rng))==static_cast<fp_t>(0.0L)) {}
465 auto c1=static_cast<fp_t>(-2.0L*re); // -2*re
466 auto c2=static_cast<fp_t>(pr_product_difference(re, re, -im, im)); // re*re+im*im
467
468 for (int j = P-2; j >= 0; --j){
469 big_coeffs_new[j] = ttmath::Big<exponent,mantissa>(std::fma(big_coeffs[j].ToDouble(), c1,
470 std::fma(big_coeffs[j+1].ToDouble(), c2, big_coeffs[j+2].ToDouble())));
471 }
472 big_coeffs_new[P-first4_complex_roots_cnt] *= ttmath::Big<exponent,mantissa>(c2) ; // first not null element
473 big_coeffs = big_coeffs_new;
474 // std::cout << "\nIM: " << im << "\n";
475 // roots[P-i*2-1] = re; // In future we can add support of complex numbers
476 // roots[P-i*2-2] = re;
477 }
478 for (int i=0; i < P+1; ++i) {
479 coefficients[i] = static_cast<fp_t>(big_coeffs[i].ToDouble());
480 }
481 return P - (N_pairs_of_complex_roots*2);
482 }
483 }
484return -1; // unreachable, means a flaw in control here
485}

504template<typename fp_t>
505int compare_roots(
506unsigned N_roots_to_check,
507unsigned N_roots_ground_truth,
508std::vector<fp_t> &roots_to_check,
509std::vector<fp_t> &roots_ground_truth,
510fp_t &max_absolute_error,
511fp_t &max_relative_error){
512 int rv = (N_roots_to_check<N_roots_ground_truth) ? PR_AT_LEAST_ONE_ROOT_LOST :
513 ( (N_roots_to_check>N_roots_ground_truth) ? PR_AT_LEAST_ONE_ROOT_IS_FAKE : PR_NUMBERS_OF_ROOTS_EQUAL );
514 long double abs = std::numeric_limits<long double >::max();
515 long double rel = std::numeric_limits<long double >::max();
516 auto size = roots_to_check.size();
517 for(int j = 0;j<size; j++)
518 for(int i = 0;i < size; i++){
519 long double absLoc = std::abs((long double)(roots_ground_truth[i])-(long double)(roots_to_check[(i + j) % size]));
520 abs = std::min(absLoc,abs);
