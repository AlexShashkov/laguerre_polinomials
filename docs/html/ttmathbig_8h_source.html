66/*
67 value = mantissa * 2^exponent
68
69 - exponent - an integer value with a sign
70 - mantissa - an integer value without a sing
71
72 mantissa must be pushed into the left side that is the highest bit from
73 mantissa must be one (of course if there's another value than zero) -- this job
74 (pushing bits into the left side) is doing by Standardizing() method
75
76 for example:
77 if we want to store value one (1) into our Big object we must:
78 - set mantissa to 1
79 - set exponent to 0
80 - set info to 0
81 - and call method Standardizing()
82*/

173 uint Standardizing()
174 {
175 if( mantissa.IsTheHighestBitSet() )
176 {
177 ClearInfoBit(TTMATH_BIG_ZERO);
178 return 0;
179 }
180
181 if( CorrectZero() )
182 return 0;
183
184 uint comp = mantissa.CompensationToLeft();
185
186 return exponent.Sub( comp );
187 }

198 bool CorrectZero()
199 {
200 if( mantissa.IsZero() )
201 {
202 SetInfoBit(TTMATH_BIG_ZERO);
203 ClearInfoBit(TTMATH_BIG_SIGN);
204 exponent.SetZero();
205
206 return true;
207 }
208 else
209 {
210 ClearInfoBit(TTMATH_BIG_ZERO);
211 }
212
213 return false;
214 }

930 uint Add(Big<exp, man> ss2, bool round = true, bool adding = true)
931 {
932 bool last_bit_set, rest_zero, do_adding, do_rounding, rounding_up;
933 uint c = 0;
934
935 if( IsNan() || ss2.IsNan() )
936 return CheckCarry(1);
937
938 if( !adding )
939 ss2.ChangeSign(); // subtracting
940
941 // (1) abs(this) will be >= abs(ss2)
942 if( SmallerWithoutSignThan(ss2) )
943 Swap(ss2);
944
945 if( ss2.IsZero() )
946 return 0;
947
948 Int<exp> exp_offset( exponent );
949 exp_offset.Sub( ss2.exponent );
950
951 if( !exp_offset.Abs() )
952 {
953 // if there is a carry in Abs it means the value in exp_offset has only the lowest bit set
954 // so the value is the smallest possible integer
955 // and its Abs would be greater than mantissa size in bits
956 // so the method AddCheckExponents would do nothing
957
958 last_bit_set = rest_zero = do_adding = do_rounding = false;
959 rounding_up = (IsSign() == ss2.IsSign());
960
961 AddCheckExponents(ss2, exp_offset, last_bit_set, rest_zero, do_adding, do_rounding);
962
963 if( do_adding )
964 c += AddMantissas(ss2, last_bit_set, rest_zero);
965
966 if( !round || !last_bit_set )
967 do_rounding = false;
968
969 if( do_rounding )
970 c += RoundHalfToEven(rest_zero, rounding_up);
971
972 if( do_adding || do_rounding )
973 c += Standardizing();
974 }
975
976
977 return CheckCarry(c);
978 }

1969
1970 if( x.IsNan() )
1971 return CheckCarry(1);
1972
1973 if( x.IsZero() )
1974 {
1975 SetOne();

1977 }

1986 // e_.mantissa.table[0] = 1;

1998 m - the value of the mantissa in range (-1,1)
1999 e_ - 2^exponent
2000
2001 e_ can be as:
2002 ...2^-2, 2^-1, 2^0, 2^1, 2^2 ...
2003 ...1/4 , 1/2 , 1 , 2 , 4 ...
2004
2005 above one e_ is integer
2006

3989 return 1;

3991 // special method for base equal 2
3992 if( conv.base == 2 )
3993 return ToString_CreateNewMantissaAndExponent_Base2(new_man, new_exp);
3994
3995 // special method for base equal 4
3996 if( conv.base == 4 )
3997 return ToString_CreateNewMantissaAndExponent_BasePow2(new_man, new_exp, 2);
3998

4008 // this = mantissa * 2^exponent
4009

4014 temp.exponent = exponent;
4015 temp.mantissa.SetOne();

4019 // if new_exp_ is positive and with fraction then we add one
4020 Big<exp+1,man> new_exp_;
4021 c += new_exp_.ToString_Log(temp, conv.base); // this logarithm isn't very complicated
4022

4024 if( !new_exp_.IsInteger() )
4025 {
4026 if( !new_exp_.IsSign() )
4027 c += new_exp_.AddOne(); // new_exp_ > 0 and with fraction

4038 c += ToString_CreateNewMantissaTryExponent<string_type, char_type>(new_man, conv, new_exp_, new_exp);

4054 const Big<exp+1,man> & new_exp_, Int<exp+1> & new_exp) const
4055 {

4058 // because 'base^new_exp' is >= '2^exponent' then
4059 // because base is >= 2 then we've got:
4060 // 'new_exp_' must be smaller or equal 'new_exp'
4061 // and we can pass it into the Int<exp> type
4062 // (in fact we're using a greater type then it'll be ok)

4064
4065 // base_ = base
4066 Big<exp+1,man> base_(conv.base);
4067

4452

4479
4480 return true; // it will be integer after rounding
4481 }
